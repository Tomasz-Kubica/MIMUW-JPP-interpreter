// Zwraca po ilu krokach w problemie Collatza otrzymamy 1 zaczynajÄ…c od x
int col(int x) {
  int counter = 0;
  while (x != 1) {
    counter = counter + 1;
    if (x % 2 == 0)
      x = x / 2;
    else
      x = 3 * x + 1;
  }
  return counter;
}

print col(1); // 0
print col(4); // 2
print col(42); // 8
